A template for StateMachines. The state is represented by a variant of structs that represent specific states. The subclasses must provide behavior for each possible specific state the state variant could be with a function for each state: behavior(const State &state) {...}.This implementation avoids lengthy switch statements, verbose expansion using polymorphic states,

#pragma once
#include "task.hpp"
#include <iostream>
#include <type_traits>
#include <variant>
namespace bfb {
template <class Concrete, typename ValidStates, std::uint32_t priority = TASK_PRIORITY_DEFAULT>
class StateMachine : public Task<priority> {
public:
StateMachine(const ValidStates &iState) : state(iState) {
}
virtual void step() final {
for (;;)
std::visit([this](const auto &s) { static_cast<Concrete *>(this)->behavior(s); }, state);
}
virtual void setState(const ValidStates &iState) {
state = iState;
}
virtual ValidStates getState() const final {
return state;
}
virtual bool operator==(const ValidStates &iState) const final {
return state.index() == iState.index();
}
virtual bool operator!=(const ValidStates &iState) const final {
return state.index() != iState.index();
}
protected:
ValidStates state;
};
} // namespace bfb
